CREATE TABLE dwh.dwa_police_report (
	customer_id numeric NULL,
	user_id numeric NULL,
	first_name varchar(100) NULL,
	last_name varchar(100) NULL,
	door_number varchar(30) NULL,
	street varchar(200) NULL,
	postal_code varchar(20) NULL,
	city varchar(100) NULL,
	state_code varchar(30) NULL,
	country varchar(100) NULL,
	address_description text NULL,
	msisdn_number varchar(100) NULL,
	subscription_start_date timestamp NULL,
	subscription_end_date timestamp NULL,
	imsi varchar(100) NULL,
	imei varchar(100) NULL,
	service_type text NULL,
	start_datetime timestamp NULL,
	end_datetime timestamp NULL,
	quantity int8 NULL,
	unit_of_quantity varchar(40) NULL,
	destination_number varchar(250) NULL,
	serving_cell_address varchar(32) NULL,
	gdr_id int8 NULL,
	call_type text NULL,
	extbillingaccountid int4 NULL,
	cdate date NULL,
	etl_date timestamptz NULL
)
PARTITION BY RANGE (cdate);
CREATE INDEX idx_police_report_customer_id ON ONLY dwh.dwa_police_report USING btree (customer_id);
